92 {

94  uint32_t base_pgentry = pfn_virt_start % 1024;
95 
96  uint32_t pg_offset = 0;
97  for (int i = base_pgtable; i < 1024 && pfn_count; i++) {
98  page_table_t *table = boot_pgtables + i;
99 
100  uint32_t pgentry_start = (i == base_pgtable) ? base_pgentry : 0;
101 
102  for (int j = pgentry_start; j < 1024 && pfn_count; j++, pfn_count--) {
103  table->pages[j].frame = pfn_phys_start + pg_offset++;
104  table->pages[j].rw = 1;
105  table->pages[j].present = 1;
106  table->pages[j].global = 0;

110  boot_pgdir.entries[i].present = 1;
111  boot_pgdir.entries[i].available = 1;
112  boot_pgdir.entries[i].frame = ((uint32_t)table) >> 12u;
113  }

119 static inline void __setup_boot_paging()
120 {
121  uint32_t kernel_base_phy_page = boot_info.kernel_phy_start >> 12U;
122  uint32_t kernel_base_virt_page = boot_info.kernel_start >> 12U;
123  // Compute the last physical page.
124  uint32_t lowmem_last_phy_page = ((uint32_t)(boot_info.lowmem_phy_end - 1)) >> 12U;
125  // Compute the number of pages.
126  uint32_t num_pages = lowmem_last_phy_page - kernel_base_phy_page + 1;
127  // Map lowmem physical pages also to their physical address (to keep bootloader working)
128  __setup_pages(0, 0, lowmem_last_phy_page);
129  // Setup kernel virtual address space + lowmem
130  __setup_pages(kernel_base_virt_page, kernel_base_phy_page, num_pages);
131 }

137 static void __get_kernel_low_high(elf_header_t *elf_hdr, uint32_t *virt_low, uint32_t *virt_high)
138 {
139  // Prepare a pointer to a program header.
140  elf_program_header_t *program_header;
141  // Compute the offset for accessing the program headers.
142  uint32_t offset = (uint32_t)elf_hdr + (uint32_t)elf_hdr->phoff;
143  // In this two variables we will store the start and end addresses of the segment.
144  uint32_t segment_start, segment_end;
145  // Iterate for each program header.
146  for (int i = 0; i < elf_hdr->phnum; i++) {
147  program_header = (elf_program_header_t *)(offset + elf_hdr->phentsize * i);
148  if (program_header->type == PT_LOAD) {
149  // Take the start and end addresses of the segment from the program header.
150  segment_start = program_header->vaddr;
151  segment_end = segment_start + program_header->memsz;
152  // Take the lowest and highest virtual address.
153  *virt_low = min(*virt_low, segment_start);
154  *virt_high = max(*virt_high, segment_end);
155  }
156  }
157 }

162 static inline uint32_t __get_address_after_modules(multiboot_info_t *header)
163 {
164  // We set by default the address to the ending physical address
165  // of the bootloader.
166  uint32_t addr = boot_info.bootloader_phy_end;
167  // Get the pointer to the mods.
168  multiboot_module_t *mod = (multiboot_module_t *)header->mods_addr;
169  for (int i = 0; (i < header->mods_count) && (i < MAX_MODULES); ++i, ++mod) {
170  addr = max(max(addr, mod->mod_start), mod->mod_end);
171  }

177 static inline void __relocate_kernel_image(elf_header_t *elf_hdr)
178 {
179  // Support variables.
180  elf_program_header_t *program_header;
181  char *kernel_start, *virtual_address, *physical_address;
182  uint32_t offset, valid_size;
183 
184  // Get the elf file starting address.
185  kernel_start = (char *)elf_hdr;
186  // Compute the offset for accessing the program headers.
187  offset = (uint32_t)kernel_start + (uint32_t)elf_hdr->phoff;
188  // Iterate over the program headers.
189  for (int i = 0; i < elf_hdr->phnum; i++) {
190  // Get the program header.
191  program_header = (elf_program_header_t *)(offset + elf_hdr->phentsize * i);
192  // Get the virtual address of the program header.
193  virtual_address = (char *)program_header->vaddr;
194  // Get the physical address of the program header.
195  physical_address = (char *)(kernel_start + program_header->offset);
196  // Move only the loadable segments.
197  if (program_header->type == PT_LOAD) {
198  // Get the valid size of the segment by taking the minimum between
199  // the size in bytes of the segment in the file image, in memory.
200  valid_size = min(program_header->filesz, program_header->memsz);
201  // Copy the physical data of the image to the corresponding virtual address.
202  for (int j = 0; j < valid_size; j++)
203  virtual_address[j] = physical_address[j];
204  // Set to 0 parts not present in memory!
205  for (int j = valid_size; j < program_header->memsz; j++)
206  virtual_address[j] = 0;
207  }
208  }
209 }

215 void boot_main(uint32_t magic, multiboot_info_t *header, uint32_t esp)
216 {
217  __debug_puts("\n[bootloader] Start...\n");
218  elf_header_t *elf_hdr = (elf_header_t *)LDVAR(kernel_bin);
219 
220  // Get the physical addresses of where the kernel starts and ends.
221  uint32_t boot_start = (uint32_t)_bootloader_start;
222  uint32_t boot_end = (uint32_t)_bootloader_end;
223 
224  // Extract the lowest and highest address of the kernel.
225  uint32_t kernel_virt_low = 0xFFFFFFFF;
226  uint32_t kernel_virt_high = 0;
227  __get_kernel_low_high(elf_hdr, &kernel_virt_low, &kernel_virt_high);
228 
229  // Initialize the boot_info_t structure.
230  __debug_puts("[bootloader] Initializing the boot_info structure...\n");
231  boot_info.magic = magic;
232  boot_info.bootloader_phy_start = boot_start;
233  boot_info.bootloader_phy_end = boot_end;
234  boot_info.kernel_start = kernel_virt_low;
235  boot_info.kernel_end = kernel_virt_high;
236  boot_info.kernel_size = kernel_virt_high - kernel_virt_low;
237  boot_info.multiboot_header = header;
238 
239  // Get the address after the modules.
240  boot_info.module_end = __get_address_after_modules(header);
241 
242  // Get the starting address of the physical pages at the end of the modules.
243  uint32_t kernel_phy_page_start = __align_rup(boot_info.module_end, PAGE_SIZE);
244  // Get the starting address of the virtual pages.
245  uint32_t kernel_virt_page_start = __align_rdown(kernel_virt_low, PAGE_SIZE);
246 
247  // Compute the absolute offset of the first virtual page, by subtracting
248  // the starting address of the virtual pages and the lowest virtual address
249  // of the kernel.
250  uint32_t kernel_page_offset = kernel_virt_page_start - kernel_virt_low;
251 
252  // If we add the offset we computed earlier to the physical address where
253  // the modules ends, we obtain the starting address of the physical memory.
254  boot_info.kernel_phy_start = kernel_phy_page_start + kernel_page_offset;
255  // The ending address of the physical memory is just the start plus the
256  // size of the kernel (virt_high - virt_low).
257  boot_info.kernel_phy_end = boot_info.kernel_phy_start + boot_info.kernel_size;
258 
259  boot_info.lowmem_phy_start = __align_rup(boot_info.kernel_phy_end, PAGE_SIZE);
260  boot_info.lowmem_phy_end = 896 * 1024 * 1024; // 896 MB of low memory max
261 
262  uint32_t lowmem_size = boot_info.lowmem_phy_end - boot_info.lowmem_phy_start;
263 
264  boot_info.lowmem_start = __align_rup(boot_info.kernel_end, PAGE_SIZE);
265  boot_info.lowmem_end = boot_info.lowmem_start + lowmem_size;
266 
267  boot_info.highmem_phy_start = boot_info.lowmem_phy_end;
268  boot_info.highmem_phy_end = header->mem_upper * 1024;
269  boot_info.stack_end = boot_info.lowmem_end;
270 
271  // Setup the page directory and page tables for the boot.
272  __debug_puts("[bootloader] Setting up paging...\n");
273  __setup_boot_paging();
274 
275  // Switch to the newly created page directory.
276  __debug_puts("[bootloader] Switching page directory...\n");
277  paging_switch_directory(&boot_pgdir);
278 
279  // Enable paging.
280  __debug_puts("[bootloader] Enabling paging...\n");
281  paging_enable();
282 
283  // Reserve space for the kernel stack at the end of lowmem.
284  boot_info.stack_base = boot_info.lowmem_end;
285  boot_info.lowmem_phy_end = boot_info.lowmem_phy_end - KERNEL_STACK_SIZE;
286  boot_info.lowmem_end = boot_info.lowmem_end - KERNEL_STACK_SIZE;
287 
288  __debug_puts("[bootloader] Relocating kernel image...\n");
289  __relocate_kernel_image(elf_hdr);
290 
291  __debug_puts("[bootloader] Calling `boot_kernel`...\n\n");
292  boot_kernel(boot_info.stack_base, elf_hdr->entry, &boot_info);
293 }
