53 enum zone_type {
54  /*
55  * ZONE_DMA is used when there are devices that are not able
56  * to do DMA to all of addressable memory (ZONE_NORMAL). Then we
57  * carve out the portion of memory that is needed for these devices.
58  * The range is arch specific.
59  *
60  * Some examples
61  *
62  * Architecture Limit
63  * ---------------------------
64  * parisc, ia64, sparc <4G
65  * s390 <2G
66  * arm Various
67  * alpha Unlimited or 0-16MB.
68  *
69  * i386, x86_64 and multiple other arches
70  * <16M.
71  */
72 
73  // /// Direct memory access.
74  //ZONE_DMA,
75 
76  /*
77  * Normal addressable memory is in ZONE_NORMAL. DMA operations can be
78  * performed on pages in ZONE_NORMAL if the DMA devices support
79  * transfers to all addressable memory.
80  */
82  ZONE_NORMAL,
83 
84  /*
85  * A memory area that is only addressable by the kernel through
86  * mapping portions into its own address space. This is for example
87  * used by i386 to allow the kernel to address the memory beyond
88  * 900MB. The kernel will set up special mappings (page
89  * table entries on i386) for each page that the kernel needs to
90  * access.
91  */
93  ZONE_HIGHMEM,
