470 {
471  // Check the permissions.
472  task_struct *task = scheduler_get_current_process();
473  // The current task is the owner.
474  if (task->uid == uid) {
475  if (!bitmask_check(mask, S_IRUSR))
476  return false;
477  if (bitmask_check(flags, O_WRONLY) && !bitmask_check(mask, S_IWUSR))
478  return false;
479  if (bitmask_check(flags, O_RDWR) && (!bitmask_check(mask, S_IRUSR) || !bitmask_check(mask, S_IWUSR)))
480  return false;
481  }
482  if (task->gid == gid) {
483  if (!bitmask_check(mask, S_IRGRP))
484  return false;
485  if (bitmask_check(flags, O_WRONLY) && !bitmask_check(mask, S_IWGRP))
486  return false;
487  if (bitmask_check(flags, O_RDWR) && (!bitmask_check(mask, S_IRGRP) || !bitmask_check(mask, S_IWGRP)))
488  return false;
489  }
490  if ((task->uid != uid) && (task->gid != gid)) {
491  if (!bitmask_check(mask, S_IROTH))
492  return false;
493  if (bitmask_check(flags, O_WRONLY) && !bitmask_check(mask, S_IWOTH))
494  return false;
495  if (bitmask_check(flags, O_RDWR) && (!bitmask_check(mask, S_IROTH) || !bitmask_check(mask, S_IWOTH)))
496  return false;
497  }
498  return true;

995 static int ext2_allocate_inode(ext2_filesystem_t *fs, unsigned preferred_group)
996 {
997  uint32_t group_index = 0, linear_index = 0, inode_index = 0;
998  // Lock the filesystem.
999  spinlock_lock(&fs->spinlock);
1000  // Allocate the cache.
1001  uint8_t *cache = kmem_cache_alloc(fs->ext2_buffer_cache, GFP_KERNEL);
1002  // Clean the cache.
1003  memset(cache, 0, fs->block_size);
1004  // Search for a free inode.
1005  if (!ext2_find_free_inode(fs, cache, &group_index, &linear_index, preferred_group)) {
1006  pr_warning("Failed to find a free inode.\n");
1007  // Unlock the filesystem.
1008  spinlock_unlock(&fs->spinlock);
1009  // Free the cache.
1010  kmem_cache_free(cache);
1011  return 0;
1012  }
1013  // Compute the inode index.
1014  inode_index = (group_index * fs->superblock.inodes_per_group) + linear_index + 1U;
1015  // Set the inode as occupied.
1016  ext2_set_bitmap_bit(cache, linear_index, ext2_block_status_occupied);
1017  // Write back the inode bitmap.
1018  ext2_write_block(fs, fs->block_groups[group_index].inode_bitmap, cache);
1019  // Free the cache.
1020  kmem_cache_free(cache);
1021  // Reduce the number of free inodes.
1022  fs->block_groups[group_index].free_inodes_count -= 1;
1023  // Update the bgdt.
1024  ext2_write_bgdt(fs);
1025  // Reduce the number of inodes inside the superblock.
1026  fs->superblock.free_inodes_count -= 1;
1027  // Update the superblock.
1028  ext2_write_superblock(fs);
1029  // Unlock the filesystem.
1030  spinlock_unlock(&fs->spinlock);
1031  // Return the inode.
1032  return inode_index;
1033 }

1041  // Lock the filesystem.
1042  spinlock_lock(&fs->spinlock);

1054  return 0;
1055  }

1273  // Return the direct block pointer.
1274  if (block_index < EXT2_INDIRECT_BLOCKS) {
1275  return inode->data.blocks.dir_blocks[block_index];
1276  }
1277  // Create a variable for the real index.
1278  uint32_t real_index = 0;
1279  // For simplicity.
1280  uint32_t p1 = fs->pointers_per_block;
1281  uint32_t p2 = fs->pointers_per_block * fs->pointers_per_block;
1282  // Allocate the cache.
1283  uint8_t *cache = kmem_cache_alloc(fs->ext2_buffer_cache, GFP_KERNEL);
1284  // Clean the cache.
1285  memset(cache, 0, fs->block_size);
1286  // Check if the index is among the indirect blocks.
1287  if (block_index < fs->indirect_blocks_index) {
1288  // Compute the indirect indices.
1289  uint32_t a = block_index - EXT2_INDIRECT_BLOCKS;
1290  // Read the indirect block (which contains pointers to the next set of blocks).
1291  ext2_read_block(fs, inode->data.blocks.indir_block, cache);
1292  // Compute the index inside the final block.
1293  real_index = ((uint32_t *)cache)[a];
1294  } else if (block_index < fs->doubly_indirect_blocks_index) {
1295  // The index is among the doubly-indirect blocks.
1296  // Compute the indirect indices.
1297  uint32_t a = block_index - EXT2_INDIRECT_BLOCKS;
1298  uint32_t b = a - p1;
1299  uint32_t c = b / p1;
1300  uint32_t d = b - (c * p1);
1301  // Read the doubly-indirect block (which contains pointers to indirect blocks).
1302  ext2_read_block(fs, inode->data.blocks.doubly_indir_block, cache);
1303  // Compute the index inside the indirect block.
1304  ext2_read_block(fs, ((uint32_t *)cache)[c], cache);
1305  // Compute the index inside the final block.
1306  real_index = ((uint32_t *)cache)[d];
1307  } else if (block_index < fs->trebly_indirect_blocks_index) {
1308  // The index is among the trebly-indirect blocks.
1309  // Compute the indirect indices.
1310  uint32_t a = block_index - EXT2_INDIRECT_BLOCKS;
1311  uint32_t b = a - p1;
1312  uint32_t c = b - p2;
1313  uint32_t d = c / p2;
1314  uint32_t e = c - (d * p2);
1315  uint32_t f = e / p1;
1316  uint32_t g = e - (f * p1);
1317  // Read the trebly-indirect block (which contains pointers to doubly-indirect blocks).
1318  ext2_read_block(fs, inode->data.blocks.trebly_indir_block, cache);
1319  // Read the doubly-indirect block (which contains pointers to indirect blocks).
1320  ext2_read_block(fs, ((uint32_t *)cache)[d], cache);
1321  // Read the indirect block (which contains pointers to the next set of blocks).
1322  ext2_read_block(fs, ((uint32_t *)cache)[f], cache);
1323  // Compute the index inside the final block.
1324  real_index = ((uint32_t *)cache)[g];
1325  } else {
1326  pr_err("We failed to retrieve the real block number of the block with index `%d`\n", block_index);
1327  }
1328  // Free the cache.
1329  kmem_cache_free(cache);
1330  // Return the real index.
1331  return real_index;
1332 }

1664 static int ext2_allocate_direntry(ext2_filesystem_t *fs, vfs_file_t *parent, const char *name, uint32_t inode_index, uint8_t file_type)
1665 {
1666  // Get the inode associated with the file.
1667  ext2_inode_t parent_inode;
1668  if (ext2_read_inode(fs, &parent_inode, parent->ino) == -1) {
1669  pr_err("Failed to read the inode (%d).\n", parent->ino);
1670  return -1;
1671  }
1672  // Check that the parent is a directory.
1673  if (!bitmask_check(parent_inode.mode, EXT2_S_IFDIR)) {
1674  pr_err("The parent inode is not a directory (ino: %d, mode: %d).\n", parent->ino, parent_inode.mode);
1675  return -1;
1676  }
1677  pr_debug("ext2_allocate_direntry(parent: \"%s\", name: \"%s\", inode: %d)\n", parent->name, name, inode_index);
1678  // Compute the rec_len for the name of the new direntry. Remember, the name
1679  // is not actually 256 chars long as specified in EXT2_NAME_LEN, that is
1680  // just a maximum.
1681  unsigned int rec_len = ext2_get_rec_len_from_name(name);
1682 
1683  pr_debug("Our directory entry looks like this:\n");
1684  pr_debug(" inode = %d\n", inode_index);
1685  pr_debug(" rec_len = %d\n", rec_len);
1686  pr_debug(" name = %s (%d)\n", name, strlen(name));
1687  pr_debug(" file_type = %d (vfs: %d)\n", EXT2_S_IFREG, DT_REG);
1688 
1689  // Allocate the cache.
1690  uint8_t *cache = kmem_cache_alloc(fs->ext2_buffer_cache, GFP_KERNEL);
1691  // Clean the cache.
1692  memset(cache, 0, fs->block_size);
1693  // Save the previous direntry.
1694  ext2_dirent_t *previous = NULL;
1695  // Iterate the directory entries.
1696  ext2_direntry_iterator_t it = ext2_direntry_iterator_begin(fs, cache, &parent_inode);
1697  for (; ext2_direntry_iterator_valid(&it); ext2_direntry_iterator_next(&it)) {
1698  // If we hit a direntry with an empty inode, that is a free direntry.
1699  if (it.direntry->inode == 0) {
1700  pr_debug("Found free direntry: %p (%d <= %d)\n", it.direntry, it.direntry->rec_len, rec_len);
1701  if (rec_len <= it.direntry->rec_len)
1702  break;
1703  }
1704  // Compute the real rec_len of the entry.
1705  uint32_t real_rec_len = ext2_get_rec_len_from_direntry(it.direntry);
1706  // If the previous direntry has a wrong rec_len (wrong because it identifies the last).
1707  if ((it.direntry->rec_len != real_rec_len) && (it.total_offset + it.direntry->rec_len == parent_inode.size)) {
1708  // Set the previous pointer.
1709  previous = it.direntry;
1710  // Fix the rec_len of the entry.
1711  it.direntry->rec_len = real_rec_len;
1712  // Move the block offset correctly.
1713  it.block_offset += real_rec_len;
1714  // Move the total offset correctly.
1715  it.total_offset += real_rec_len;
1716  // Clean the pointer to the direntry inside the iterator.
1717  it.direntry = NULL;
1718  // Stop here.
1719  break;
1720  }
1721  }
1722  if (it.direntry) {
1723  pr_debug("We need to replace the direntry: %p\n", it.direntry);
1724  // Clean the previous name.
1725  memset(it.direntry->name, 0, it.direntry->name_len);
1726  // Set the inode.
1727  it.direntry->inode = inode_index;
1728  // Set the new name length,
1729  it.direntry->name_len = strlen(name);
1730  // Set the new name.
1731  memcpy(it.direntry->name, name, it.direntry->name_len);
1732  // Set the file type.
1733  it.direntry->file_type = file_type;
1734 
1735  if (ext2_write_inode_block(fs, &parent_inode, parent->ino, it.block_index, cache) == -1) {
1736  pr_err("Failed to update the block of the father directory.\n");
1737  // Free the cache.
1738  kmem_cache_free(cache);
1739  return -1;
1740  }
1741  // Free the cache.
1742  kmem_cache_free(cache);
1743  return 0;
1744  } else if ((it.block_offset + rec_len) >= fs->block_size) {
1745  pr_debug("We need a new direntry, and a new block (%d + %d >= %d).\n", it.block_offset, rec_len, fs->block_size);
1746  it.block_index += 1;
1747  if (ext2_allocate_inode_block(fs, &parent_inode, parent->ino, it.block_index) == -1) {
1748  pr_err("Failed to allocate a new block for an inode.\n");
1749  // Free the cache.
1750  kmem_cache_free(cache);
1751  return -1;
1752  }
1753  it.block_offset = 0;
1754  parent_inode.size += fs->block_size;
1755  if (ext2_write_inode(fs, &parent_inode, parent->ino) == -1) {
1756  pr_err("Failed to update the inode of the father directory.\n");
1757  // Free the cache.
1758  kmem_cache_free(cache);
1759  return -1;
1760  }
1761  } else if (previous) {
1762  pr_debug("We need a new direntry, from the same block (after \"%s\").\n", previous->name);
1763  }
1764  pr_debug(" total_offset = %d\n", it.total_offset);
1765  pr_debug(" block_offset = %d\n", it.block_offset);
1766 
1767  ext2_dirent_t *new_direntry = (ext2_dirent_t *)((uintptr_t)cache + it.block_offset);
1768 
1769  new_direntry->inode = inode_index;
1770  new_direntry->rec_len = fs->block_size - it.block_offset;
1771  new_direntry->name_len = strlen(name);
1772  new_direntry->file_type = file_type;
1773  memcpy(new_direntry->name, name, strlen(name));
1774 
1775  if (ext2_write_inode_block(fs, &parent_inode, parent->ino, it.block_index, cache) == -1) {
1776  pr_err("Failed to update the block of the father directory.\n");
1777  // Free the cache.
1778  kmem_cache_free(cache);
1779  return -1;
1780  }
1781  // Free the cache.
1782  kmem_cache_free(cache);
1783  return 0;
1784 }

2847 {
2848  // Create the ext2 filesystem.
2849  ext2_filesystem_t *fs = kmalloc(sizeof(ext2_filesystem_t));
2850  // Clean the memory.
2851  memset(fs, 0, sizeof(ext2_filesystem_t));
2852  // Initialize the filesystem spinlock.
2853  spinlock_init(&fs->spinlock);
2854  // Initialize the list of opened files.
2855  list_head_init(&fs->opened_files);
2856  // Set the pointer to the block device.
2857  fs->block_device = block_device;
2858  // Read the superblock.
2859  if (ext2_read_superblock(fs) == -1) {
2860  pr_err("Failed to read the superblock table at 1024.\n");
2861  // Free just the filesystem.
2862  goto free_filesystem;
2863  }
2864  // Check the superblock magic number.
2865  if (fs->superblock.magic != EXT2_SUPERBLOCK_MAGIC) {
2866  pr_err("Wrong magic number, it is not an EXT2 filesystem.\n");
2867  ext2_dump_superblock(&fs->superblock);
2868  // Free just the filesystem.
2869  goto free_filesystem;
2870  }
2871  // Compute the volume size.
2872  fs->block_size = 1024U << fs->superblock.log_block_size;
2873  // Initialize the buffer cache.
2874  fs->ext2_buffer_cache = kmem_cache_create(
2875  "ext2_buffer_cache",
2876  fs->block_size,
2877  fs->block_size,
2878  GFP_KERNEL,
2879  NULL,
2880  NULL);
2881  // Compute the maximum number of inodes per block.
2882  fs->inodes_per_block_count = fs->block_size / sizeof(ext2_inode_t);
2883  // Compute the number of blocks per block. This value is mostly used for
2884  // inodes.
2885  // If you check inside the inode structure you will find the `blocks_count`
2886  // field. A 32-bit value representing the total number of 512-bytes blocks
2887  // reserved to contain the data of this inode, regardless if these blocks
2888  // are used or not. The block numbers of these reserved blocks are contained
2889  // in the `block` array.
2890  // Since this value represents 512-byte blocks and not file system blocks,
2891  // this value should not be directly used as an index to the `block` array.
2892  // Rather, the maximum index of the `block` array should be computed from
2893  // inode->blocks_count / ((1024 << superblock->log_block_size) / 512)
2894  // or once simplified:
2895  // inode->blocks_count / (2 << superblock->log_block_size)
2896  // Now we just need to precompute the right part.
2897  fs->blocks_per_block_count = fs->block_size / 512U;
2898  // Compute the number of block pointers per block.
2899  fs->pointers_per_block = fs->block_size / 4U;
2900  // Compute the index of indirect blocks.
2901  fs->indirect_blocks_index = EXT2_INDIRECT_BLOCKS + fs->pointers_per_block;
2902  fs->doubly_indirect_blocks_index = EXT2_INDIRECT_BLOCKS + fs->pointers_per_block * (fs->pointers_per_block + 1);
2903  fs->trebly_indirect_blocks_index = fs->doubly_indirect_blocks_index +
2904  (fs->pointers_per_block * fs->pointers_per_block) * (fs->pointers_per_block + 1);
2905  // Compute the number of block groups.
2906  fs->block_groups_count = fs->superblock.blocks_count / fs->superblock.blocks_per_group;
2907  if (fs->superblock.blocks_per_group * fs->block_groups_count < fs->superblock.blocks_count) {
2908  fs->block_groups_count += 1;
2909  }
2910  // The block group descriptor table starts on the first block following the
2911  // superblock. This would be the second block for 2KiB and larger block file systems.
2912  if (fs->block_size > K) {
2913  fs->bgdt_start_block = 1;
2914  } else {
2915  // However, it would be the third block on a 1KiB block file system.
2916  fs->bgdt_start_block = 2;
2917  }
2918  // The block group descriptor table ends a certain amount of blocks.
2919  fs->bgdt_end_block = fs->bgdt_start_block + ((sizeof(ext2_group_descriptor_t) * fs->block_groups_count) / fs->block_size) + 1;
2920  // Compute the length in blocks of the BGDT.
2921  fs->bgdt_length = fs->bgdt_end_block - fs->bgdt_start_block;
2922 
2923  // Now, we have the size of a block, calculate the location of the Block
2924  // Group Descriptor Table (BGDT). The BGDT is located directly after the
2925  // superblock, so obtain the block of the superblock first.
2926  fs->block_groups = kmalloc(fs->block_size * fs->bgdt_length);
2927  if (fs->block_groups == NULL) {
2928  pr_err("Failed to allocate memory for the block buffer.\n");
2929  // Free just the filesystem.
2930  goto free_filesystem;
2931  }
2932 
2933  // Try to read the BGDT.
2934  if (ext2_read_bgdt(fs) == -1) {
2935  pr_err("Failed to read the BGDT.\n");
2936  // Free the block_groups and the filesystem.
2937  goto free_block_groups;
