217 static void __set_mode(vga_mode_t *vga_mode)
218 {
219  unsigned char *ptr = &vga_mode->misc;
220 
221  // Write SEQUENCER regs.
222  outportb(MISC_WRITE, *ptr);
223  ++ptr;
224 
225  // Write SEQUENCER regs.
226  for (unsigned i = 0; i < MODE_NUM_SEQ_REGS; i++) {
227  outportb(SC_INDEX, i);
228  outportb(SC_DATA, *ptr);
229  ++ptr;
230  }
231 
232  // Wnlock CRTC registers.
233  outportb(CRTC_INDEX, 0x03);
234  outportb(CRTC_DATA, inportb(CRTC_DATA) | 0x80);
235  outportb(CRTC_INDEX, 0x11);
236  outportb(CRTC_DATA, inportb(CRTC_DATA) & ~0x80);
237 
238  // Make sure they remain unlocked.
239  ptr[0x03] |= 0x80;
240  ptr[0x11] &= ~0x80;
241 
242  // write CRTC regs.
243  for (unsigned i = 0; i < MODE_NUM_CRTC_REGS; i++) {
244  outportb(CRTC_INDEX, i);
245  outportb(CRTC_DATA, *ptr);
246  ++ptr;
247  }
248 
249  // write GRAPHICS CONTROLLER regs.
250  for (unsigned i = 0; i < MODE_NUM_GC_REGS; i++) {
251  outportb(GC_INDEX, i);
252  outportb(GC_DATA, *ptr);
253  ++ptr;
254  }
255 
256  // write ATTRIBUTE CONTROLLER regs.
257  for (unsigned i = 0; i < MODE_NUM_AC_REGS; i++) {
258  inportb(INPUT_STATUS_READ);
259  outportb(AC_INDEX, i);
260  outportb(AC_WRITE, *ptr);
261  ++ptr;
262  }
263 
264  // lock 16-color palette and unblank display.
265  (void)inportb(INPUT_STATUS_READ);
266  outportb(AC_INDEX, 0x20);
267 }

316 static void __write_font(unsigned char *buf, unsigned font_height)
317 {
318  unsigned char seq2, seq4, gc4, gc5, gc6;
319  unsigned i;
320 
321  /* save registers
322 __set_plane() modifies GC 4 and SEQ 2, so save them as well */
323  outportb(SC_INDEX, 2);
324  seq2 = inportb(SC_DATA);
325 
326  outportb(SC_INDEX, 4);
327  seq4 = inportb(SC_DATA);
328  /* turn off even-odd addressing (set flat addressing)
329 assume: chain-4 addressing already off */
330  outportb(SC_DATA, seq4 | 0x04);
331 
332  outportb(GC_INDEX, 4);
333  gc4 = inportb(GC_DATA);
334 
335  outportb(GC_INDEX, 5);
336  gc5 = inportb(GC_DATA);
337  /* turn off even-odd addressing */
338  outportb(GC_DATA, gc5 & ~0x10);
339 
340  outportb(GC_INDEX, 6);
341  gc6 = inportb(GC_DATA);
342  /* turn off even-odd addressing */
343  outportb(GC_DATA, gc6 & ~0x02);
344  /* write font to plane P4 */
345  __set_plane(2);
346  /* write font 0 */
347  for (i = 0; i < 256; i++) {
348  memcpy((char *)buf, (char *)(i * 32 + 0xA0000), font_height);
349  buf += font_height;
350  }
351  /* restore registers */
352  outportb(SC_INDEX, 2);
353  outportb(SC_DATA, seq2);
354  outportb(SC_INDEX, 4);
355  outportb(SC_DATA, seq4);
356  outportb(GC_INDEX, 4);
357  outportb(GC_DATA, gc4);
358  outportb(GC_INDEX, 5);
359  outportb(GC_DATA, gc5);
360  outportb(GC_INDEX, 6);
361  outportb(GC_DATA, gc6);
362 }

537 {

541  tmp = x0;
542  x0 = y0;

544 
545  tmp = x1;
546  x1 = y1;
547  y1 = tmp;

551  x0 = x1;
552  x1 = tmp;

554  tmp = y0;
555  y0 = y1;
556  y1 = tmp;
557  }

559  int deltay = abs(y1 - y0);

564  ystep = 1;

566  ystep = -1;

571  driver->ops->write_pixel(x, y, color);

575  error = error + deltax;
576  }
577  }
