238 static void __set_mode(vga_mode_t *vga_mode)
239 {
240  unsigned char *ptr = &vga_mode->misc;
241 
242  // Write SEQUENCER regs.
243  outportb(MISC_WRITE, *ptr);
244  ++ptr;
245 
246  // Write SEQUENCER regs.
247  for (unsigned i = 0; i < MODE_NUM_SEQ_REGS; i++) {
248  outportb(SC_INDEX, i);
249  outportb(SC_DATA, *ptr);
250  ++ptr;
251  }
252 
253  // Wnlock CRTC registers.
254  outportb(CRTC_INDEX, 0x03);
255  outportb(CRTC_DATA, inportb(CRTC_DATA) | 0x80);
256  outportb(CRTC_INDEX, 0x11);
257  outportb(CRTC_DATA, inportb(CRTC_DATA) & ~0x80);
258 
259  // Make sure they remain unlocked.
260  ptr[0x03] |= 0x80;
261  ptr[0x11] &= ~0x80;
262 
263  // write CRTC regs.
264  for (unsigned i = 0; i < MODE_NUM_CRTC_REGS; i++) {
265  outportb(CRTC_INDEX, i);
266  outportb(CRTC_DATA, *ptr);
267  ++ptr;
268  }
269 
270  // write GRAPHICS CONTROLLER regs.
271  for (unsigned i = 0; i < MODE_NUM_GC_REGS; i++) {
272  outportb(GC_INDEX, i);
273  outportb(GC_DATA, *ptr);
274  ++ptr;
275  }
276 
277  // write ATTRIBUTE CONTROLLER regs.
278  for (unsigned i = 0; i < MODE_NUM_AC_REGS; i++) {
279  inportb(INPUT_STATUS_READ);
280  outportb(AC_INDEX, i);
281  outportb(AC_WRITE, *ptr);
282  ++ptr;
283  }
284 
285  // lock 16-color palette and unblank display.
286  (void)inportb(INPUT_STATUS_READ);
287  outportb(AC_INDEX, 0x20);
288 }

337 static void __write_font(unsigned char *buf, unsigned font_height)
338 {
339  unsigned char seq2, seq4, gc4, gc5, gc6;
340  unsigned i;
341 
342  /* save registers
343 __set_plane() modifies GC 4 and SEQ 2, so save them as well */
344  outportb(SC_INDEX, 2);
345  seq2 = inportb(SC_DATA);
346 
347  outportb(SC_INDEX, 4);
348  seq4 = inportb(SC_DATA);
349  /* turn off even-odd addressing (set flat addressing)
350 assume: chain-4 addressing already off */
351  outportb(SC_DATA, seq4 | 0x04);
352 
353  outportb(GC_INDEX, 4);
354  gc4 = inportb(GC_DATA);
355 
356  outportb(GC_INDEX, 5);
357  gc5 = inportb(GC_DATA);
358  /* turn off even-odd addressing */
359  outportb(GC_DATA, gc5 & ~0x10);
360 
361  outportb(GC_INDEX, 6);
362  gc6 = inportb(GC_DATA);
363  /* turn off even-odd addressing */
364  outportb(GC_DATA, gc6 & ~0x02);
365  /* write font to plane P4 */
366  __set_plane(2);
367  /* write font 0 */
368  for (i = 0; i < 256; i++) {
369  memcpy((char *)buf, (char *)(i * 32 + 0xA0000), font_height);
370  buf += font_height;
371  }
372  /* restore registers */
373  outportb(SC_INDEX, 2);
374  outportb(SC_DATA, seq2);
375  outportb(SC_INDEX, 4);
376  outportb(SC_DATA, seq4);
377  outportb(GC_INDEX, 4);
378  outportb(GC_DATA, gc4);
379  outportb(GC_INDEX, 5);
380  outportb(GC_DATA, gc5);
381  outportb(GC_INDEX, 6);
382  outportb(GC_DATA, gc6);
383 }

575 {

577  int dy = abs(y1 - y0), sy = sign(y1 - y0);

759  driver = &driver_640_480_16;
760  // Load the color palette.
761  __load_palette(ansi_16_palette, 16);
762  // Set the font.
763  driver->font = &font_8x14;
764 #elif defined(VGA_MODE_720_480_16) // 90x60

766  __set_mode(&_mode_720_480_16);

769  // Load the color palette.

783  vga_enable = true;
784 }

800 {

807 {
808  unsigned char color = (_cursor_state = (_cursor_state == 0)) * _color;
809  for (unsigned cy = 0; cy < driver->font->height; ++cy)
810  for (unsigned cx = 0; cx < driver->font->width; ++cx)
811  vga_draw_pixel(_x + cx, _y + cy, color);

814 void vga_putc(int c)
815 {
816  if (_cursor_state)
817  __vga_clear_cursor();
818  // If the character is '\n' go the new line.
819  if (c == '\n') {
820  vga_new_line();
821  } else if ((c >= 0x20) && (c <= 0x7E)) {
822  vga_draw_char(_x, _y, c, _color);
823  if ((_x += driver->font->width) >= driver->width)
824  vga_new_line();
825  } else {
826  return;
827  }
828 }

875  // Go back at the beginning of the line.
876  _x = 0;

878  _y = 0;

880  }
