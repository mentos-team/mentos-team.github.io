711 
712  outportb(dev->control_base, 0x00);
713  outportb(dev->io_reg.hddevsel, 0xe0 | (dev->slave << 4));
714  ata_io_wait(dev);
715  outportb(dev->io_reg.feature, 0x00);
716 
717  outportb(dev->io_reg.sector_count, 0);
718  outportb(dev->io_reg.lba_lo, (lba & 0xff000000) >> 24);
719  outportb(dev->io_reg.lba_mid, (lba & 0xff00000000) >> 32);
720  outportb(dev->io_reg.lba_hi, (lba & 0xff0000000000) >> 40);
721 
722  outportb(dev->io_reg.sector_count, 1);
723  outportb(dev->io_reg.lba_lo, (lba & 0x000000ff) >> 0);
724  outportb(dev->io_reg.lba_mid, (lba & 0x0000ff00) >> 8);
725  outportb(dev->io_reg.lba_hi, (lba & 0x00ff0000) >> 16);
726 
727  while (1) {
728  uint8_t status = inportb(dev->io_reg.status);
729  if (!bit_check(status, ata_status_bsy) && bit_check(status, ata_status_rdy))
730  break;
731  }
732 
733  // Write the READ_DMA to the command register (0xC8)
734  outportb(dev->io_reg.command, ata_dma_command_read);
735 
736  ata_io_wait(dev);
737 
738  outportb(dev->bmr.command, 0x08 | 0x01);
739 
740  while (1) {
741  int status = inportb(dev->bmr.status);
742  int dstatus = inportb(dev->io_reg.status);
743  if (!(status & 0x04)) {
744  continue;
745  }
746  if (!bit_check(dstatus, ata_status_bsy)) {
747  break;
748  }
749  }
750 
751  // Copy from DMA buffer to output buffer.
752  memcpy(buffer, dev->dma_start, ATA_DMA_SIZE);
753 
754  // Inform device we are done.
755  outportb(dev->bmr.status, inportb(dev->bmr.status) | 0x04 | 0x02);
756 
757  spinlock_unlock(&ata_lock);
758 }

760 static void ata_device_write_sector(ata_device_t *dev, uint32_t lba, uint8_t *buffer)
761 {
762  spinlock_lock(&ata_lock);
763 
764  // Copy the buffer over to the DMA area
765  memcpy(dev->dma_start, buffer, ATA_DMA_SIZE);
766 
767  // Reset bus master register's command register
768  outportb(dev->bmr.command, 0);
769 
770  // Set prdt
771  outportl(dev->bmr.prdt, dev->dma_prdt_phys);
772 
773  // Enable error, irq status.
774  outportb(dev->bmr.status, inportb(dev->bmr.status) | 0x04 | 0x02);
775 
776  // Select drive
777  ata_wait(dev, 0);
778  outportb(dev->io_reg.hddevsel, 0xe0 | dev->slave << 4 | (lba & 0x0f000000) >> 24);
779  ata_wait(dev, 0);
780 
781  // Set sector counts and LBAs
782  outportb(dev->io_reg.feature, 0x00);
783  outportb(dev->io_reg.sector_count, 1);
784  outportb(dev->io_reg.lba_lo, (lba & 0x000000ff) >> 0);
785  outportb(dev->io_reg.lba_mid, (lba & 0x0000ff00) >> 8);
786  outportb(dev->io_reg.lba_hi, (lba & 0x00ff0000) >> 16);
787  ata_wait(dev, 0);
788 
789  // Notify that we are starting DMA writing.
790  outportb(dev->io_reg.command, ata_dma_command_write);
791 
792  // Start DMA Writing.
793  outportb(dev->bmr.command, 0x1);
794 
795  // Wait for dma write to complete.
796  while (1) {
797  int status = inportb(dev->bmr.status);
798  int dstatus = inportb(dev->io_reg.status);
799  if (!(status & 0x04)) {
800  continue;
801  }
802  if (!(dstatus & 0x80)) {
803  break;
804  }
805  }
806 
807  // Inform device we are done.
808  outportb(dev->bmr.status, inportb(dev->bmr.status) | 0x04 | 0x02);
809 
810  spinlock_unlock(&ata_lock);
811 }

894 static ssize_t ata_read(vfs_file_t *file, char *buffer, off_t offset, size_t size)
895 {
896  pr_debug("ata_read(file: 0x%p, buffer: 0x%p, offest: %8d, size: %8d)\n", file, buffer, offset, size);
897  // Prepare a static support buffer.
898  static char support_buffer[ATA_SECTOR_SIZE];
899  // Get the device from the VFS file.
900  ata_device_t *dev = (ata_device_t *)file->device;
901  // Check the device.
902  if (dev == NULL) {
903  kernel_panic("Device not set.");
904  }
905 
906  if ((dev->type == ata_dev_type_pata) || (dev->type == ata_dev_type_sata)) {
907  uint32_t start_block = offset / ATA_SECTOR_SIZE;
908  uint32_t start_offset = offset % ATA_SECTOR_SIZE;
909  uint32_t end_block = (offset + size - 1) / ATA_SECTOR_SIZE;
910  uint32_t end_offset = (offset + size - 1) % ATA_SECTOR_SIZE;
911  uint32_t prefix_size = (ATA_SECTOR_SIZE - start_offset);
912  uint32_t postfix_size = (offset + size) % ATA_SECTOR_SIZE;
913  uint32_t max_offset = ata_max_offset(dev);
914  uint32_t x_offset = 0;
915 
916  // Check if with the offset we are exceeding the size.
917  if (offset > max_offset) {
918  pr_warning("The offset is exceeding the disk size (%d > %d)\n", offset, max_offset);
919  return 0;
920  }
921 
922  // Check if we are going to reading over the size.
923  if ((offset + size) > max_offset) {
924  size = max_offset - offset;
925  }
926 
927  if (start_offset) {
928  ata_device_read_sector(dev, start_block, (uint8_t *)support_buffer);
929  memcpy(buffer, (void *)((uintptr_t)support_buffer + start_offset), prefix_size);
930  x_offset += prefix_size;
931  ++start_block;
932  }
933 
934  if (postfix_size && (start_block <= end_block)) {
935  ata_device_read_sector(dev, end_block, (uint8_t *)support_buffer);
936  memcpy((void *)((uintptr_t)buffer + size - postfix_size), support_buffer, postfix_size);
937  --end_block;
938  }
939 
940  while (start_block <= end_block) {
941  ata_device_read_sector(dev, start_block, (uint8_t *)((uintptr_t)buffer + x_offset));
942  x_offset += ATA_SECTOR_SIZE;
943  ++start_block;
944  }
945  } else if ((dev->type == ata_dev_type_patapi) || (dev->type == ata_dev_type_satapi)) {
946  pr_warning("ATAPI and SATAPI drives are not currently supported.\n");
947  size = -EPERM;
948  }
949  return size;
950 }
