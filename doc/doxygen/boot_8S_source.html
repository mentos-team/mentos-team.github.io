7 bits 32 ; All instructions should be 32-bit.
8 extern boot_main ; The start point of our C code
9 
10 ; The magic field should contain this.
11 MULTIBOOT_HEADER_MAGIC equ 0x1BADB002
12 ; This should be in %eax.
13 MULTIBOOT_BOOTLOADER_MAGIC equ 0x2BADB002
14 
15 ; = Specify what GRUB should PROVIDE =========================================
16 ; Align the kernel and kernel modules on i386 page (4KB) boundaries.
17 MULTIBOOT_PAGE_ALIGN equ 0x00000001
18 ; Provide the kernel with memory information.
19 MULTIBOOT_MEMORY_INFO equ 0x00000002
20 ; Must pass video information to OS.
21 MULTIBOOT_VIDEO_MODE equ 0x00000004
22 ; -----------------------------------------------------------------------------
23 
24 ; This is the flag combination that we prepare for Grub
25 ; to read at kernel load time.
26 MULTIBOOT_HEADER_FLAGS equ (MULTIBOOT_PAGE_ALIGN | MULTIBOOT_MEMORY_INFO | MULTIBOOT_VIDEO_MODE)
27 ; Grub reads this value to make sure it loads a kernel
28 ; and not just garbage.
29 MULTIBOOT_CHECKSUM equ -(MULTIBOOT_HEADER_MAGIC + MULTIBOOT_HEADER_FLAGS)
30 
31 LOAD_MEMORY_ADDRESS equ 0x00000000
32 ; reserve (1024*1024) for the stack on a doubleword boundary
33 KERNEL_STACK_SIZE equ 0x100000
34 
35 ; -----------------------------------------------------------------------------
36 ; SECTION (multiboot_header)
37 ; -----------------------------------------------------------------------------
38 section .multiboot_header
39 align 4
40 ; This is the GRUB Multiboot header.
41 multiboot_header:
42  ; magic
43  dd MULTIBOOT_HEADER_MAGIC
44  ; flags
45  dd MULTIBOOT_HEADER_FLAGS
46  ; checksum
47  dd MULTIBOOT_CHECKSUM
48 
49 ; -----------------------------------------------------------------------------
50 ; SECTION (data)
51 ; -----------------------------------------------------------------------------
52 ; section .data nobits
53 ; align 4096
54 
55 ; -----------------------------------------------------------------------------
56 ; SECTION (text)
57 ; -----------------------------------------------------------------------------
58 section .text
59 global boot_entry
60 boot_entry:
61  ; Clear interrupt flag [IF = 0]; 0xFA
62  cli
63  ; To set up a stack, we simply set the esp register to point to the top of
64  ; our stack (as it grows downwards).
65  mov esp, stack_top
66  ; pass the initial ESP
67  push esp
68  ; pass Multiboot info structure
69  push ebx
70  ; pass Multiboot magic number
71  push eax
72  ; Call the boot_main() function inside boot.c
73  call boot_main
74  ; Set interrupt flag [IF = 1]; 0xFA
75  ; Clear interrupts and hang if we return from boot_main
76  cli
77 hang:
78  hlt
79  jmp hang
80 
81 global boot_kernel
82 boot_kernel:
83  mov edx, [esp + 4] ; stack_pointer
84  mov ebx, [esp + 8] ; entry
85  mov eax, [esp + 12] ; boot info
86  mov esp, edx ; set stack pointer
87  push eax ; push the boot info
88  call ebx ; call the kernel main
89

95 align 16
96 
97 global stack_bottom
98 stack_bottom:
99  resb KERNEL_STACK_SIZE

101 global stack_top
102 stack_top:
