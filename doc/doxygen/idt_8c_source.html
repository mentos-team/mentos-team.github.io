131 static inline void __idt_set_gate(uint8_t index, interrupt_handler_t handler, uint16_t options, uint8_t seg_sel)
132 {
133  uintptr_t base_prt = (uintptr_t)handler;
134  // Assign the base values.
135  idt_table[index].offset_low = (base_prt & 0xFFFFu);
136  idt_table[index].offset_high = (base_prt >> 16u) & 0xFFFFu;
137  // Set the other fields.
138  idt_table[index].reserved = 0x00;
139  idt_table[index].seg_selector = seg_sel;
140  idt_table[index].options = options | IDT_PADDING;
141 }
