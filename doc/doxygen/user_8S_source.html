7 ; Enter userspace (ring3) (from Ring 0, namely Kernel)
8 ; Usage: enter_userspace(uintptr_t location, uintptr_t stack);
9 ; On stack
10 ; | stack | [ebp + 0x0C] ARG1
11 ; | location | [ebp + 0x08] ARG0
12 ; | return address | [ebp + 0x04]
13 ; | EBP | [ebp + 0x00]
14 ; | SS |
15 ; | ESP |
16 ; | EFLAGS |
17 ; | CS |
18 ; | EIP |
19 
20 ; We can use the following macros to access the arguments ONLY AFTER 0x23 is
21 ; pushed onto the stack, in fact, the first argument is after 0x08 because
22 ; we just pushed first `ebp` and then `0x23`.
23 %define ARG0 [ebp + 0x08] ; Argument 0
24 %define ARG1 [ebp + 0x0C] ; Argument 1
25 %define ARG2 [ebp + 0x10] ; Argument 2
26 %define ARG3 [ebp + 0x14] ; Argument 3
27 %define ARG4 [ebp + 0x18] ; Argument 4
28 
29 ; -----------------------------------------------------------------------------
30 ; SECTION (text)
31 ; -----------------------------------------------------------------------------
32 section .text

34 global enter_userspace ; Allows the C code to call enter_userspace(...).

36 enter_userspace:

38  push ebp ; Save current ebp
39  mov ebp, esp ; open a new stack frame

41  ;==== Segment selector =====================================================
42  mov ax, 0x23
43  mov ds, ax
44  mov es, ax
45  mov fs, ax
46  mov gs, ax
47  ; we don't need to worry about SS. it's handled by iret
48  ;---------------------------------------------------------------------------

50  ; We have to prepare the following stack before executing iret
51  ; SS --> Segment selector
52  ; ESP --> Stack address
53  ; EFLAGS --> CPU state flgas
54  ; CS --> Code segment
55  ; EIP --> Entry point
56  ;
57 
58  ;==== User data segmenet with bottom 2 bits set for ring3 ?=================
59  push 0x23 ; push SS on Kernel's stack
60  ;---------------------------------------------------------------------------
61 
62  ;==== (ESP) Stack address ==================================================
63  mov eax, ARG1 ; get uintptr_t stack
64  push eax ; push process's stack address on Kernel's stack
65  ;---------------------------------------------------------------------------
66 
67  ;==== (EFLAGS) =============================================================
68  pushf ; push EFLAGS into Kernel's stack
69  pop eax ; pop EFLAGS into eax
70  or eax, 0x200 ; enable interrupt
71  push eax ; push new EFLAGS on Kernel's stack
72  ;---------------------------------------------------------------------------
73 
74  ;==== (CS) Code Segment ====================================================
75  push 0x1b ;
76  ;---------------------------------------------------------------------------
77 
78  ;==== (EIP) Entry point ====================================================
79  mov eax, ARG0 ; get uintptr_t location
80  push eax ; push uintptr_t location on Kernel's stack
81  ;---------------------------------------------------------------------------

83  iret ; interrupt return
