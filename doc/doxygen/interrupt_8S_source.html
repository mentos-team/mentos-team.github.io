9 %macro IRQ 2
10  global IRQ_%1
11  IRQ_%1:
12  cli ; disable interrupt line
13  ; A normal ISR stub that pops a dummy error code to keep a
14  ; uniform stack frame
15  push 0
16  push %2 ; irq number
17  jmp irq_common
18 %endmacro

43 irq_common:
44  ;==== Save CPU registers ===================================================
45  ; when an irq occurs, the following registers are already pushed on stack:
46  ; eip, cs, eflags, useresp, ss
47 
48  ; Save registers: eax, ecx, edx, ebx, esp, ebp, esi, edi
49  pusha
50 
51  ; Save segment registers
52  push ds
53  push es
54  push fs
55  push gs
56  ;---------------------------------------------------------------------------
57 
58  ;==== ensure we are using kernel data segment ==============================
59  mov ax, 0x10
60  mov ds, ax
61  mov es, ax
62  mov fs, ax
63  mov gs, ax
64  cld
65  ;---------------------------------------------------------------------------
66 
67  ;==== Call the interrupt handler ===========================================
68  ; The argument passed is a pointer to an Interrupt_State struct,
69  ; which describes the stack layout for all interrupts.
70  push esp
71  call irq_handler
72  add esp, $4 ; remove esp from stack
73  ;---------------------------------------------------------------------------
74 
75  ;==== Restore registers ====================================================
76  ; restore segment registers
77  pop gs
78  pop fs
79  pop es
80  pop ds
81 
82  ; restore registers: eax, ecx, edx, ebx, esp, ebp, esi, edi
83  popa
84  ;---------------------------------------------------------------------------
85 
86  ; Cleanup error code and IRQ #
87  add esp, $8
88 
89  ; return to process
90  iret ; pops 5 things at once:
91  ; CS, EIP, EFLAGS, SS, and ESP
