39 // PS/2 Controller Configuration Byte
40 // Bit | Meaning
41 // 0 | First PS/2 port interrupt (1 = enabled, 0 = disabled)
42 // 1 | Second PS/2 port interrupt (1 = enabled, 0 = disabled, only if 2 PS/2 ports supported)
43 // 2 | System Flag (1 = system passed POST, 0 = your OS shouldn't be running)
44 // 3 | Should be zero
45 // 4 | First PS/2 port clock (1 = disabled, 0 = enabled)
46 // 5 | Second PS/2 port clock (1 = disabled, 0 = enabled, only if 2 PS/2 ports supported)
47 // 6 | First PS/2 port translation (1 = enabled, 0 = disabled)
48 // 7 | Must be zero

143 int ps2_initialize()
144 {
145  unsigned char status, response;
146  bool_t dual;
147 
148  status = __ps2_get_controller_status();
149  pr_debug("Status : %s (%3d | %02x)\n", dec_to_binary(status, 8), status, status);
150 
151  // ========================================================================
152  // Step 1: Disable Devices
153  // So that any PS/2 devices can't send data at the wrong time and mess up
154  // your initialisation; start by sending a command 0xAD and command 0xA7 to
155  // the PS/2 controller. If the controller is a "single channel" device, it
156  // will ignore the "command 0xA7".
157 
158  pr_debug("Disabling first port...\n");
159  __ps2_disable_first_port();
160 
161  pr_debug("Disabling second port...\n");
162  __ps2_disable_second_port();
163 
164  // ========================================================================
165  // Step 2: Flush The Output Buffer
166  // Sometimes (e.g. due to interrupt controlled initialisation causing a lost
167  // IRQ) data can get stuck in the PS/2 controller's output buffer. To guard
168  // against this, now that the devices are disabled (and can't send more data
169  // to the output buffer) it can be a good idea to flush the controller's
170  // output buffer. There's 2 ways to do this - poll bit 0 of the Status
171  // Register (while reading from IO Port 0x60 if/when bit 0 becomes set), or
172  // read from IO Port 0x60 without testing bit 0. Either way should work (as
173  // you're discarding the data and don't care what it was).
174 
175  pr_debug("Flushing the output buffer...\n");
176  ps2_read();
177 
178  // ========================================================================
179  // Step 3: Set the Controller Configuration Byte
180  // Because some bits of the Controller Configuration Byte are "unknown",
181  // this means reading the old value (command 0x20), changing some bits, then
182  // writing the changed value back (command 0x60). You want to disable all
183  // IRQs and disable translation (clear bits 0, 1 and 6).
184  // While you've got the Configuration Byte, test if bit 5 was set. If it was
185  // clear, then you know it can't be a "dual channel" PS/2 controller
186  // (because the second PS/2 port should be disabled).
187 
188  // Get the status.
189  status = __ps2_get_controller_status();
190  pr_debug("Disable IRQs and translation...\n");
191  // Clear bits 0, 1 and 6.
192  bit_clear_assign(status, 0);
193  bit_clear_assign(status, 1);
194  // We want to keep the translation from Set 2/3 to Set 1, active.
195  //bit_clear_assign(status, 6);
196  __ps2_set_controller_status(status);
197  pr_debug("Status : %s (%3d | %02x)\n", dec_to_binary(status, 8), status, status);
198 
199  // ========================================================================
200  // Step 4: Perform Controller Self Test
201  // To test the PS/2 controller, send command 0xAA to it. Then wait for its
202  // response and check it replied with 0x55. Note: this can reset the PS/2
203  // controller on some hardware (tested on a 2016 laptop). At the very least,
204  // the Controller Configuration Byte should be restored for compatibility
205  // with such hardware. You can either determine the correct value yourself
206  // based on the above table or restore the value read before issuing 0xAA.
207 
208  // Send 0xAA to the controller.
209  __ps2_write_command(PS2_CTRL_TEST_CONTROLLER);
210  // Read the response.
211  if ((response = ps2_read()) == PS2_TEST_FAIL1) {
212  pr_err("Self-test failed : 0x%02x\n", response);
213  return 1;
214  }
215  // Restore the value read before issuing 0xAA.
216  __ps2_set_controller_status(status);
217 
218  // ========================================================================
219  // Step 5: Determine If There Are 2 Channels
220  // Enable the second PS/2 port and read the Controller Configuration Byte
221  // again. Now, bit 5 of the Controller Configuration Byte should be clear -
222  // if it's set then you know it can't be a "dual channel" PS/2 controller
223  // (because the second PS/2 port should be enabled). If it is a dual channel
224  // device, disable the second PS/2 port again.
225 
226  // Enable the second port.
227  __ps2_enable_second_port();
228  // Read the status.
229  status = __ps2_get_controller_status();
230  // Check if it is a dual channel PS/2 device.
231  if ((dual = !bit_check(status, 5))) {
232  pr_debug("Recognized a `dual channel` PS/2 controller...\n");
233  __ps2_disable_second_port();
234  } else {
235  pr_debug("Recognized a `single channel` PS/2 controller...\n");
236  }
237 
238  // ========================================================================
239  // Step 6: Perform Interface Tests
240  // This step tests the PS/2 ports. Use command 0xAB to test the first PS/2
241  // port, then check the result. Then (if it's a "dual channel" controller)
242  // use command 0xA9 to test the second PS/2 port, then check the result.
243  // At this stage, check to see how many PS/2 ports are left. If there aren't
244  // any that work you can just give up (display errors and terminate the PS/2
245  // Controller driver).
246  // Note: If one of the PS/2 ports on a dual PS/2 controller fails, then you
247  // can still keep using/supporting the other PS/2 port.
248 
249  __ps2_write_command(PS2_CTRL_P1_TEST);
250  if ((response = ps2_read()) && (response >= 0x01) && (response <= 0x04)) {
251  pr_err("Interface test failed on first port : %02x\n", response);
252  pr_err("Reason: %s.\n", __ps2_get_response_error_message(response));
253  return 1;
254  }
255  // If it is a dual channel, check the second port.
256  if (dual) {
257  __ps2_write_command(PS2_CTRL_P2_TEST);
258  if ((response = ps2_read()) && (response >= 0x01) && (response <= 0x04)) {
259  pr_err("Interface test failed on second port : %02x\n", response);
260  pr_err("Reason: %s.\n", __ps2_get_response_error_message(response));
261  return 1;
262  }
263  }
264 
265  // ========================================================================
266  // Step 7: Enable Devices
267  // Enable any PS/2 port that exists and works.
268  // If you're using IRQs (recommended), also enable interrupts for any
269  // (usable) PS/2 ports in the Controller Configuration Byte (set bit 0 for
270  // the first PS/2 port, and/or bit 1 for the second PS/2 port, then set it
271  // with command 0x60).
272 
273  // Enable the first port.
274  __ps2_enable_first_port();
275  // Enable the second port.
276  if (dual)
277  __ps2_enable_second_port();
278  // Get the status.
279  status = __ps2_get_controller_status();
280  pr_debug("Status : %s (%3d | %02x)\n", dec_to_binary(status, 8), status, status);
281  // Set bit 0 for the first port.
282  bit_set_assign(status, 0);
283  if (dual) {
284  // Set bit 1 for the second port.
285  bit_set_assign(status, 1);
286  }
287  __ps2_set_controller_status(status);
288 
289  // ========================================================================
290  // Step 8: Reset Devices
291  // All PS/2 devices should support the "reset" command (which is a command
292  // for the device, and not a command for the PS/2 Controller).
293  // To send the reset, just send the byte 0xFF to each (usable) device. The
294  // device/s will respond with 0xFA (success) or 0xFC (failure), or won't
295  // respond at all (no device present). If your code supports "hot-plug PS/2
296  // devices" (see later), then you can assume each device is "not present"
297  // and let the hot-plug code figure out that the device is present if/when
298  // 0xFA or 0xFC is received on a PS/2 port.
299 
300  // Reset first port.
301  __ps2_write_first_port(0xFF);
302  // Wait for `command acknowledged`.
303  if ((response = ps2_read()) != PS2_ACK) {
304  pr_err("Failed to reset first PS/2 port: %d\n", response);
305  return 1;
306  }
307  // Wait for `self test successful`.
308  if ((response = ps2_read()) != PS2_TEST_SUCCESS) {
309  pr_err("Failed to reset first PS/2 port: %d\n", response);
310  return 1;
311  }
312 
313  // Reset second port.
314  __ps2_write_second_port(0xFF);
315  // Wait for `command acknowledged`.
316  if ((response = ps2_read()) != PS2_ACK) {
317  pr_err("Failed to reset first PS/2 port: %d\n", response);
318  return 1;
319  }
320  // Wait for `self test successful`.
321  if ((response = ps2_read()) != PS2_TEST_SUCCESS) {
322  pr_err("Failed to reset first PS/2 port: %d\n", response);
323  return 1;
324  }
325 
326  // Get the final status.
327  status = __ps2_get_controller_status();
328  pr_debug("Status : %s (%3d | %02x)\n", dec_to_binary(status, 8), status, status);
329 
330  return 0;
331 }
