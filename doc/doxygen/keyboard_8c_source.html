119 void keyboard_isr(pt_regs *f)
120 {
121  unsigned int scancode;
122  (void)f;
123 
124  if (!(inportb(0x64U) & 1U)) {
125  return;
126  }
127 
128  // Take scancode from the port.
129  scancode = ps2_read();
130  if (scancode == 0xE0) {
131  scancode = (scancode << 8U) | ps2_read();
132  }
133 
134  // Get the keypad number, of num-lock is disabled. Otherwise, initialize to -1;
135  int keypad_fun_number = !bitmask_check(kflags, KBD_NUM_LOCK) ? get_keypad_number(scancode) : -1;
136 
137  // If the key has just been released.
138  if (scancode == KEY_LEFT_SHIFT) {
139  bitmask_set_assign(kflags, KBD_LEFT_SHIFT);
140  pr_debug("Press(KBD_LEFT_SHIFT)\n");
141  } else if (scancode == KEY_RIGHT_SHIFT) {
142  bitmask_set_assign(kflags, KBD_RIGHT_SHIFT);
143  pr_debug("Press(KBD_RIGHT_SHIFT)\n");
144  } else if (scancode == KEY_LEFT_CONTROL) {
145  bitmask_set_assign(kflags, KBD_LEFT_CONTROL);
146  pr_debug("Press(KBD_LEFT_CONTROL)\n");
147  } else if (scancode == KEY_RIGHT_CONTROL) {
148  bitmask_set_assign(kflags, KBD_RIGHT_CONTROL);
149  pr_debug("Press(KBD_RIGHT_CONTROL)\n");
150  } else if (scancode == KEY_LEFT_ALT) {
151  bitmask_set_assign(kflags, KBD_LEFT_ALT);
152  keyboard_push_front(scancode << 16u);
153  pr_debug("Press(KBD_LEFT_ALT)\n");
154  } else if (scancode == KEY_RIGHT_ALT) {
155  bitmask_set_assign(kflags, KBD_RIGHT_ALT);
156  keyboard_push_front(scancode << 16u);
157  pr_debug("Press(KBD_RIGHT_ALT)\n");
158  } else if (scancode == (KEY_LEFT_SHIFT | CODE_BREAK)) {
159  bitmask_clear_assign(kflags, KBD_LEFT_SHIFT);
160  pr_debug("Release(KBD_LEFT_SHIFT)\n");
161  } else if (scancode == (KEY_RIGHT_SHIFT | CODE_BREAK)) {
162  bitmask_clear_assign(kflags, KBD_RIGHT_SHIFT);
163  pr_debug("Release(KBD_RIGHT_SHIFT)\n");
164  } else if (scancode == (KEY_LEFT_CONTROL | CODE_BREAK)) {
165  bitmask_clear_assign(kflags, KBD_LEFT_CONTROL);
166  pr_debug("Release(KBD_LEFT_CONTROL)\n");
167  } else if (scancode == (KEY_RIGHT_CONTROL | CODE_BREAK)) {
168  bitmask_clear_assign(kflags, KBD_RIGHT_CONTROL);
169  pr_debug("Release(KBD_RIGHT_CONTROL)\n");
170  } else if (scancode == (KEY_LEFT_ALT | CODE_BREAK)) {
171  bitmask_clear_assign(kflags, KBD_LEFT_ALT);
172  pr_debug("Release(KBD_LEFT_ALT)\n");
173  } else if (scancode == (KEY_RIGHT_ALT | CODE_BREAK)) {
174  bitmask_clear_assign(kflags, KBD_RIGHT_ALT);
175  pr_debug("Release(KBD_RIGHT_ALT)\n");
176  } else if (scancode == KEY_CAPS_LOCK) {
177  bitmask_flip_assign(kflags, KBD_CAPS_LOCK);
178  keyboard_update_leds();
179  pr_debug("Toggle(KBD_CAPS_LOCK)\n");
180  } else if (scancode == KEY_NUM_LOCK) {
181  bitmask_flip_assign(kflags, KBD_NUM_LOCK);
182  keyboard_update_leds();
183  pr_debug("Toggle(KBD_NUM_LOCK)\n");
184  } else if (scancode == KEY_SCROLL_LOCK) {
185  bitmask_flip_assign(kflags, KBD_SCROLL_LOCK);
186  keyboard_update_leds();
187  pr_debug("Toggle(KBD_SCROLL_LOCK)\n");
188  } else if (scancode == KEY_BACKSPACE) {
189  keyboard_push_front('\b');
190  pr_debug("Press(KEY_BACKSPACE)\n");
191  } else if (scancode == KEY_DELETE) {
192  keyboard_push_front(0x7F);
193  pr_debug("Press(KEY_DELETE)\n");
194  } else if ((scancode == KEY_ENTER) || (scancode == KEY_KP_RETURN)) {
195  keyboard_push_front('\n');
196  pr_debug("Press(KEY_ENTER)\n");
197  } else if ((scancode == KEY_PAGE_UP) || (keypad_fun_number == 9)) {
198  keyboard_push_front(scancode);
199  pr_debug("Press(KEY_PAGE_UP)\n");
200  } else if ((scancode == KEY_PAGE_DOWN) || (keypad_fun_number == 2)) {
201  keyboard_push_front(scancode);
202  pr_debug("Press(KEY_PAGE_DOWN)\n");
203  } else if ((scancode == KEY_UP_ARROW) || (keypad_fun_number == 8)) {
204  pr_debug("Press(KEY_UP_ARROW)\n");
205  keyboard_push_front('\033');
206  keyboard_push_front('[');
207  keyboard_push_front('A');
208  } else if ((scancode == KEY_DOWN_ARROW) || (keypad_fun_number == 2)) {
209  pr_debug("Press(KEY_DOWN_ARROW)\n");
210  keyboard_push_front('\033');
211  keyboard_push_front('[');
212  keyboard_push_front('B');
213  } else if ((scancode == KEY_RIGHT_ARROW) || (keypad_fun_number == 6)) {
214  pr_debug("Press(KEY_RIGHT_ARROW)\n");
215  keyboard_push_front('\033');
216  keyboard_push_front('[');
217  keyboard_push_front('C');
218  } else if ((scancode == KEY_LEFT_ARROW) || (keypad_fun_number == 4)) {
219  pr_debug("Press(KEY_LEFT_ARROW)\n");
220  keyboard_push_front('\033');
221  keyboard_push_front('[');
222  keyboard_push_front('D');
223  } else if ((scancode == KEY_HOME) || (keypad_fun_number == 7)) {
224  pr_debug("Press(KEY_HOME)\n");
225  keyboard_push_front('\033');
226  keyboard_push_front('[');
227  keyboard_push_front('H');
228  } else if ((scancode == KEY_END) || (keypad_fun_number == 1)) {
229  pr_debug("Press(KEY_END)\n");
230  keyboard_push_front('\033');
231  keyboard_push_front('[');
232  keyboard_push_front('F');
233  } else if (scancode == KEY_ESCAPE) {
234  // Nothing to do.
235  } else if (keypad_fun_number == 5) {
236  // Nothing to do.
237  } else if (!(scancode & CODE_BREAK)) {
238  // Get the current keymap.
239  const keymap_t *keymap = get_keymap(scancode);
240  // Get the specific keymap.
241  int character = 0;
242  if (!bitmask_check(kflags, KBD_LEFT_SHIFT | KBD_RIGHT_SHIFT) != !bitmask_check(kflags, KBD_CAPS_LOCK)) {
243  keyboard_push_front(keymap->shift);
244  } else if ((get_keymap_type() == KEYMAP_IT) && bitmask_check(kflags, KBD_RIGHT_ALT) && (bitmask_check(kflags, KBD_LEFT_SHIFT | KBD_RIGHT_SHIFT))) {
245  keyboard_push_front(keymap->alt);
246  } else if (bitmask_check(kflags, KBD_RIGHT_ALT)) {
247  keyboard_push_front(keymap->alt);
248  } else if (bitmask_check(kflags, KBD_LEFT_CONTROL | KBD_RIGHT_CONTROL)) {
249  keyboard_push_front(keymap->ctrl);
250  } else {
251  keyboard_push_front(keymap->normal);
252  }
253  }
254  pic8259_send_eoi(IRQ_KEYBOARD);
255 }

257 void keyboard_update_leds()
258 {
259  // Handle scroll_loc & num_loc & caps_loc.
260  bitmask_check(kflags, KBD_SCROLL_LOCK) ? (ledstate |= 1) : (ledstate ^= 1);
261  bitmask_check(kflags, KBD_NUM_LOCK) ? (ledstate |= 2) : (ledstate ^= 2);
262  bitmask_check(kflags, KBD_CAPS_LOCK) ? (ledstate |= 4) : (ledstate ^= 4);
263 
264  // Write on the port.
265  outportb(0x60, 0xED);
266  outportb(0x60, ledstate);
267 }
